```bash
npm install --save react-top-loading-bar
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-top-loading-bar'
import 'react-top-loading-bar/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
